var ArrayMap = function() {
this.keys = [];
this.values = [];
};
ArrayMap.prototype = {
get : function(key) {
return this.values[key];
},
put : function(key, val) {
if (this.values[key] == null) {
this.keys.push(key);
}
this.values[key] = val;
},
remove : function(key) {
if (this.values[key] == null) {
return;
}
for ( var m in this.keys) {
if (this.keys[m] == key) {
this.keys.splice(m, 1);
}
}
},
each : function(fn) {
if (typeof fn != 'function') {
return;
}
for ( var m in this.keys) {
fn(this.keys[m], this.get(this.keys[m]), m);
}
},
sizes : function() {
return this.keys.length;
}
};
